shiftify = (arrayish, start, end) ->

hex = (bytes) ->
  h = bytes.map (b) ->
    if b < 10
      "0" + b.toString(16)
    else
      b.toString(16)
  h.join("")

pack = (pattern, value) ->
  if pattern.type == "f"
    if pattern.bits == 32
      ieee754.toIEEE754Single value
    else
      ieee754.toIEEE754Double value
  else
    value
